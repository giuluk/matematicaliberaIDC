\newcommand{\radicedidue}{% 
  % Quadrato di lato 1 adagiato su un asse cartesiano con diagonale.
  \assex{-.3}{+1.7}{0}
  \draw 
    (0, 0) node[below]{\small$0$} rectangle ++(1, 1) 
    (1, 0) node[below]{\small$1$} 
    (0, .5) node[left]{\small$1$} 
    (1.4142, 0) node[below]{\small$\phantom{\sqrt{2}}$}
    (.5, .5) node[above left=-5pt]{\small$\sqrt{2}$};
  \draw [thick](0, 0) --++(1, 1);
}

\newcommand{\radicediduea}{% 
  \disegno[18]{\radicedidue}
}

\newcommand{\radicedidueb}{% 
  \disegno[18]{
    \radicedidue
    \centerarc[thick](0, 0)(0:45:1.4142)
    \draw (1.4142, 0) node[below]{\small$\sqrt{2}$};
  }
}

\newcommand{\riunionesegni}{% 
  \disegno{
    \trespolo*{6}{-2}
      {-1/x -5, -2/x +2}
      {2/-2, 4/+5}
    \foreach \x/\y in {4./-.6, 2./-1.6}
      \cerchietto{\x}{\y}
    \foreach \x/\s in {1/-, 3/-, 5/+}
      \node at (\x, -1) [above] {\(\s\)};
    \foreach \x/\s in {1/-, 3/+, 5/+}
      \node at (\x, -2) [above] {\(\s\)};
  }
}

\newcommand {\condradicalib}{% 
  \disegno{
    \trespolo*{6}{-3}
             {-1/Num, -2/Den, -3/Fra}
             {2/-1, 4/+1}
    \foreach \x/\y in {4/-.6}
      \cerchietto{\x}{\y}
    \foreach \x/\y in {2/-1.6}
      \crocetta{\x}{\y}
    \foreach \x/\s in {1/-, 3/-, 5/+}
      \node at (\x, -1) [above] {\(\s\)};
    \foreach \x/\s in {1/-, 3/+, 5/+}
      \node at (\x, -2) [above] {\(\s\)};
    \foreach \x/\s in {1/+, 3/-, 5/+}
      \node at (\x, -3) [above] {\(\s\)};
    \ray{-3}{0}{2}{}{white}
    \ray{-3}{6}{4}{}{blue}
  }
}
